Gulp


gulp.task('refactor', function () {
    return gulp.src('folder/**/*.js') // Нужные файлы для рефакторинга
        .pipe(RefactoringPlugin()) // Рефакторим код
        .pipe(gulp.dest('./')) // Кладем на место
})

Если что-то пошло не так, то 


$ git reset --hard
